
@(parameterNames: TraversableOnce[(String, Int)])

package uk.co.bgch.retrofit.parameters

object ParameterCodes {

    def nameForCode(code: Long): Option[String] = {
        code match {
          @for((name, value) <- parameterNames) {
            case @{value}L => Some("@name")
          }
          case _ => None
        }
    }

    val values: Seq[Long] = Seq(
        @parameterNames.map(_._2).mkString(",")
    )
}
